from pyscript import display from js import document class SimpleAgent: percepts = ["Sun", "Rain", "Cloudy", "Storm"] @staticmethod def weatherAgent(i): if SimpleAgent.percepts[i] == "Sun": return "Wear Sunglasses" elif SimpleAgent.percepts[i] in ["Rain", "Cloudy"]: return "Carry Umbrella" elif SimpleAgent.percepts[i] == "Storm": return "Stay Home" else: return "Normal Day" def run_agent(event): i = int(document.getElementById("weather").value) action = SimpleAgent.weatherAgent(i) document.getElementById("output").value = "" display(action, target="output") document.getElementById("run").onclick = run_agent
